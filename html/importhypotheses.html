ImportHypotheses

Short Description   

Imports a hypotheses file created with the ExportHypotheses function.

Additional Information   

-

Definition   

def ImportHypotheses( file = "cfdmsh_hps" ):

Argument(s)   

Name Description Type GUI selection [?] Selection by name [?] Recursive [?] Default value
file The name of the file to read. String - - - "cfdmsh_hps"

Returned Value(s)   

dim Value [?] single Value [?] Type Number Name
- - - - -

Example   

from cfdmsh import * # To adapt to the cfdmsh installation method

box = geompy.MakeBoxDXDYDZ(200, 200, 200)
AddToStudy(box, 'box')

mesh = smesh.Mesh(box)

algo = mesh.Segment()
hypo1 = algo.LocalLength(34.641, [], 1e-07)

algo = mesh.Triangle(algo = smeshBuilder.NETGEN_2D)
hypo2 = algo.Parameters()
hypo2.SetMaxSize(34.641)
hypo2.SetMinSize(0.34641)

algo = mesh.Tetrahedron()
hypo3 = algo.MaxElementVolume(41569.2)

ExportHypotheses([hypo1, hypo2, hypo3], "hypo_file_example")
ImportHypotheses("hypo_file_example")
